#### 문제 : <a href="https://programmers.co.kr/learn/courses/30/lessons/1836">프로그래머스level3_리틀 프렌즈 사천성</a>

#### 문제풀이

> 우선 각 문자의 두가지 포인트를 저장해놓고 문자 순서가 빠른것 부터 진행해야하기 때문에 하나의 문자가 사라지면 다시 처음부터 문자를 탐색해갔다.
>
> 문자는 하나의 포인트를 기준으로 총 8가지 방향이 있으며 한번이상으로 꺾지 못하기 때문에 사실상 간단한 구현문제다. 이동을 해주면서 다른 포인트에 도달한다면 지울수 있는것이고 못하면 다음 문자를 탐색하면 된다.

<br/>

```java
package programmers.level3;

public class 리틀프렌즈사천성 {
	// 0 아래쪽, 1 위쪽, 2 오른쪽, 3 왼쪽
	static int[] dx = { 1, -1, 0, 0 };
	static int[] dy = { 0, 0, 1, -1 };
	static char[][] map;
	static int x, y;
	static char c;

	public static void main(String[] args) {
		int m = 3;
		int n = 3;
		String[] board = {"DBA", "C*A", "CDB"};
		System.out.println(solution(m, n, board));
	}

	static String solution(int m, int n, String[] board) {
		map = new char[m][n];
		// 순서대로 해야한다.
		int[] visit = new int[91];
		Point[][] list = new Point[91][2];
		int amount = 0;

		for (int i = 0; i < m; i++) {
			for (int j = 0; j < n; j++) {
				char temp = board[i].charAt(j);
				if (temp >= 65 && 90 >= temp) {
					list[temp][visit[temp]++] = new Point(i, j);
					amount++;
				}

				map[i][j] = temp;
			}
		}

		StringBuilder sb = new StringBuilder();
		int cnt = 0;
		while (cnt < amount) {
			boolean isOk = false;
			for (int i = 65; i < visit.length; i++) { // 알파벳 돌면서 남아있는것을 찾는다.
				if (visit[i] == 2) { // 남아있으면
					if (check(list[i][0], list[i][1])) { // 할 수 있는지 체크해서 할 수 있으면 지운다.
						sb.append((char) i);
						isOk = true;
						visit[i] = 0;
						map[list[i][0].x][list[i][0].y] = '.';
						map[list[i][1].x][list[i][1].y] = '.';
						break;
					}
				}
			}
			
			// 쭉 돌았는데 더이상 못없애면
			if (!isOk) {
				return "IMPOSSIBLE";
			}

			cnt += 2;
		}

		return sb.toString();
	}

	static boolean check(Point p1, Point p2) {
		x = p1.x;
		y = p1.y;
		c = map[p1.x][p1.y];
		// p1이 기준이고 p2가 어디에 위치해있는지 확인해서 로직을 짜야한다.
		// 도착이 가능하면 도착한 두지점을 '.'으로 비워줘야한다.
		// 8가지를 결정해야된다. 우선 4방향 직선인경우와 4방면으로 나눠서 대각선에 위치한 경우
		if (x < p2.x) {
			if (y < p2.y) {
				// 4
				// 오른쪽 아래쪽
				if (move(p2, 2) && move(p2, 0)) {
					return true;
				}

				// 초기화
				x = p1.x;
				y = p1.y;
				// 아래쪽 오른쪽
				if (move(p2, 0) && move(p2, 2)) {
					return true;
				}

			} else if (y > p2.y) {
				// 3

				// 왼쪽 아래
				if (move(p2, 3) && move(p2, 0)) {
					return true;
				}

				x = p1.x;
				y = p1.y;

				// 아래 왼쪽
				if (move(p2, 0) && move(p2, 3)) {
					return true;
				}
			} else {
				// 7
				if (move(p2, 0)) {
					return true;
				}
			}
		} else if (x > p2.x) {
			if (y < p2.y) {
				// 2

				// 위쪽, 오른쪽
				if (move(p2, 1) && move(p2, 2)) {
					return true;
				}

				x = p1.x;
				y = p1.y;

				// 오른쪽, 위쪽
				if (move(p2, 2) && move(p2, 1)) {
					return true;
				}
			} else if (y > p2.y) {
				// 1

				// 위쪽, 왼쪽
				if (move(p2, 1) && move(p2, 3)) {
					return true;
				}

				x = p1.x;
				y = p1.y;

				// 왼쪽, 위쪽
				if (move(p2, 3) && move(p2, 1)) {
					return true;
				}
			} else {
				// 5
				// 위쪽
				if (move(p2, 1)) {
					return true;
				}
			}
		} else { // x == p2.x
			if (y > p2.y) {
				// 6
				if (move(p2, 3)) {
					return true;
				}
				// 왼쪽
			} else { // y < p2.y
				// 8
				// 오른쪽
				if (move(p2, 2)) {
					return true;
				}
			}
		}

		return false;
	}

	static boolean move(Point p2, int d) {

		while (true) {

			x += dx[d];
			y += dy[d];
			
			if(map[x][y] == c) {
				return true;
			}
			
			if (map[x][y] != '.') { // 길이 아니면 못가
				return false;
			}
			
			if (d < 2) { // 위아래로 움직이는 경우
				if (x == p2.x) {
					return true;
				}
			} else { // 오른쪽 왼쪽으로 움직이는 경우
				if (y == p2.y) {
					return true;
				}
			}
		}
	}

	static class Point {
		int x, y;

		public Point(int x, int y) {
			super();
			this.x = x;
			this.y = y;
		}
	}
}
```
